CREATE TABLE RESERVATION_TB(
RECORDID VARCHAR2(14) PRIMARY KEY,
CUSTOMERNAME VARCHAR2(40),
BOOKINGTYPE VARCHAR2(30),
BOOKING_DATE DATE,
CONFIRMATIONNO VARCHAR2(20),
AMOUNT NUMBER,
REMARKS VARCHAR2(100)
);

CREATE SEQUENCE RESERVATION_SEQ
START WITH 10
MAXVALUE 99
INCREMENT BY 1;

SELECT * FROM RESERVATION_TB;
